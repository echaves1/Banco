--a.	Crie e compile uma função chamada GET_JOB para retornar um título de trabalho.
CREATE OR REPLACE FUNCTION rm77230.get_job (jobid IN jobs.job_id%type)
RETURN jobs.job_title%type IS title jobs.job_title%type;
BEGIN

    SELECT job_title
    INTO title
    FROM jobs
    WHERE job_id = jobid;
    RETURN title;
    
END rm77230.get_job;
