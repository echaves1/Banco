-- Criar Procedure
CREATE OR REPLACE PROCEDURE format_phone (phone_no IN OUT VARCHAR2) IS 

BEGIN 
  phone_no := '(' || SUBSTR(phone_no, 1,3) ||
              ')' || SUBSTR(phone_no, 4,3) ||
              '-' || SUBSTR(phone_no, 7);
END format_phone;
/

-- Invocar procedure
var phone_no VARCHAR2(15)
EXECUTE :phone_no := '0000000000' 
PRINT phone_no
EXECUTE format_phone (:phone_no)
PRINT phone_no
